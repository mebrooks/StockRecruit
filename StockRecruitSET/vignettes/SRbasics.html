Continuous Hockey-Stick Example

Simulate data

S = simS(n=100, pars=c(60,200))
g=1 #assume this for 
R = simR(S=S, shape="contHockey", pars=c(.5, 100, g), varlog=.1)
plot(S,R)

Fit a model to the data

mod=fitSRCurve(S, R, shape="contHockey", g=g)
sdr=sdreport(mod)
summary(sdr)
##              Estimate Std. Error
## log_beta  -0.68541343 0.01942939
## log_delta  4.59782871 0.02472511
## log_sd    -2.33840239 0.07071064
## sd         0.09648166 0.00682228

Estimate Blim

calcBlim(S, R)
## [1] 99.40991

Plot the fit

Srange=seq(min(S), max(S), length=100)
Rfit=simR(S=Srange,
        shape="contHockey", 
        pars=c(exp(mod$env$last.par[c('log_beta','log_delta')]), g)
)
plot(S, R)
lines(Srange, Rfit, col=2, lwd=3)
abline(v = calcBlim(S, R), lty=2)

Ricker Example

Simulate data

S = simS(n=100, pars=c(1,200))
R = simR(S=S, shape="Ricker", pars=c(exp(1), 1/100), varlog=0.1)
plot(S,R)

Fit a model to the data

mod=fitSRCurve(S, R, shape="Ricker")
sdr=sdreport(mod)
summary(sdr)
##           Estimate  Std. Error
## log_a   0.98184839 0.018125266
## log_b  -4.62881958 0.015475703
## log_sd -2.41445254 0.070710664
## sd      0.08941628 0.006322684

Plot the fit

Srange=seq(min(S), max(S), length=100)
Rfit=simR(S=Srange,
        shape="Ricker", 
        pars=c(exp(mod$env$last.par[c('log_a','log_b')]))
)
plot(S, R)
lines(Srange, Rfit, col=2, lwd=3)
