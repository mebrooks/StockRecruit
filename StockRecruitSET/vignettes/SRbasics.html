Continuous Hockey-Stick Example

Simulate data

S = simS(n=100, pars=c(60,200))
g=1 #assume this for 
R = simR(S=S, shape="contHockey", pars=c(1, 100, g), varlog=.1)
plot(S,R)

Fit a model to the data

mod=fitSRCurve(S, R, shape="contHockey", g=g)
sdr=sdreport(mod)
summary(sdr)
##          Estimate  Std. Error
## beta    1.0176999 0.021191480
## delta  99.0856717 2.381661563
## log_sd -2.2609145 0.070710652
## sd      0.1042551 0.007371946

Plot the fit

Srange=seq(min(S), max(S), length=100)
Rfit=simR(S=Srange,
        shape="contHockey", 
        pars=c(mod$env$last.par[c('beta','delta')], g)
)
plot(S, R)
lines(Srange, Rfit, col=2, lwd=3)

Ricker Example

Simulate data

S = simS(n=100, pars=c(1,200))
R = simR(S=S, shape="Ricker", pars=c(exp(1), 1/100), varlog=0.1)
plot(S,R)

Fit a model to the data

mod=fitSRCurve(S, R, shape="Ricker")
sdr=sdreport(mod)
summary(sdr)
##          Estimate   Std. Error
## a       2.8169193 0.0534939830
## b       0.0101770 0.0001774576
## log_sd -2.2799733 0.0707106612
## sd      0.1022869 0.0072327773

Plot the fit

Srange=seq(min(S), max(S), length=100)
Rfit=simR(S=Srange,
        shape="Ricker", 
        pars=c(mod$env$last.par[c('a','b')])
)
plot(S, R)
lines(Srange, Rfit, col=2, lwd=3)

S = simS(n=100, pars=c(1,200))
R = simR(S=S, shape="BevertonHolt", pars=c(1.5456836,  0.0108868), varlog=0.1)

mod=fitSRCurve(S, R, shape="BevertonHolt")
